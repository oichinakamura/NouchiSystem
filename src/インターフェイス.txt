Function ConvDateToAge()
	Dim dtBirthday
	Dim nAge
	If Not IsDate(Param) Then
		ReturnValue = " - 歳"
	ElseIf CDate(Param) = 0 Then
		ReturnValue = " - 歳"
	Else
		dtBirthday = CDate(Param)
		nAge = DateDiff("yyyy", dtBirthday, Now())
		If DateAdd("yyyy", nAge , dtBirthday) > Now() Then nAge = nAge - 1
		ReturnValue = nAge & "歳"
	End If
End Function
